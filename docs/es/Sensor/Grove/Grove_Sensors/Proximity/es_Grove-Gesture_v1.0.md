  Serial.println("INIT OK");

{

    {
     Serial.println("Forward");
     delay(GES_QUIT_TIME);
    }
    else if(data == GES_BACKWARD_FLAG) 
    {
     Serial.println("Backward");
     delay(GES_QUIT_TIME);
    }
    else
    {
     Serial.println("Right");
    }          
    break;

    delay(GES_ENTRY_TIME);

    {
     Serial.println("Forward");
     delay(GES_QUIT_TIME);
    }
    else if(data == GES_BACKWARD_FLAG) 
    {
     Serial.println("Backward");
     delay(GES_QUIT_TIME);
    }
    else
    {
     Serial.println("Left");
    }          
    break;
   case GES_UP_FLAG:
    delay(GES_ENTRY_TIME);

    if(data == GES_FORWARD_FLAG) 
    {
     Serial.println("Forward");
     delay(GES_QUIT_TIME);
    }

    {
     Serial.println("Backward");
     delay(GES_QUIT_TIME);
    }

    {
     Serial.println("Up");
    }

     Serial.println("Forward");

    }

    {
     Serial.println("Backward");
     delay(GES_QUIT_TIME);
    }
    else
    {
     Serial.println("Down");
    }          
    break;

    Serial.println("Forward");
    delay(GES_QUIT_TIME);

    Serial.println("Backward");
    delay(GES_QUIT_TIME);

    Serial.println("Clockwise");
    break;

    Serial.println("anti-clockwise");

    paj7620ReadReg(0x44, 1, &data1);
    if (data1 == GES_WAVE_FLAG)

     Serial.println("wave");
    }
    break;
  }
 }

 def init(self):
  time.sleep(.001)
  self.paj7620SelectBank(self.BANK0)
  self.paj7620SelectBank(self.BANK0)
  
  data0 = self.paj7620ReadReg(0, 1)[0]
  data1 = self.paj7620ReadReg(1, 1)[0]
  if self.debug:
   print("data0:",data0,"data1:",data1)
  if data0 != 0x20  :#or data1 <> 0x76:
   print("Error with sensor")
   #return 0xff
  if data0 == 0x20:
   print("wake-up finish.")
   
  for i in range(len(self.initRegisterArray)):
   self.paj7620WriteReg(self.initRegisterArray[i][0],self.initRegisterArray[i][1])
  
  self.paj7620SelectBank(self.BANK0)
  
  print("Paj7620 initialize register finished.")
